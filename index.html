Project 3: Numerical Integration

This project demonstrates numerical integration using the trapezoidal rule to estimate the area under a curve.

View Code
# Python code for numerical integration (trapezoidal rule) import numpy as np # Function to integrate def f(x): return x**2 # Trapezoidal rule function def trapezoidal_rule(a, b, n): x = np.linspace(a, b, n) y = f(x) return np.trapz(y, x) # Estimate the integral of f(x) = x^2 from 0 to 1 integral_value = trapezoidal_rule(0, 1, 1000) print(f"Estimated integral value: {integral_value}")

Project 4: Simple Harmonic Motion in Python

This project simulates the motion of a mass-spring system undergoing simple harmonic motion using Python.

View Code
# Python code for simple harmonic motion import numpy as np import matplotlib.pyplot as plt # Constants k = 1 # spring constant (N/m) m = 1 # mass (kg) omega = np.sqrt(k/m) # angular frequency # Time array time = np.linspace(0, 10, 100) # Initial conditions x0 = 1 # initial displacement (m) v0 = 0 # initial velocity (m/s) # Solution to the SHM equation: x(t) = x0 * cos(omega * t) position = x0 * np.cos(omega * time) # Plotting position vs time plt.plot(time, position) plt.title('Position vs Time (Simple Harmonic Motion)') plt.xlabel('Time (s)') plt.ylabel('Position (m)') plt.show()

# Python code for linear regression import numpy as np import matplotlib.pyplot as plt from sklearn.linear_model import LinearRegression # Generate sample data X = np.random.rand(100, 1) * 10 # Random data for X (independent variable) y = 2.5 * X + np.random.randn(100, 1) * 2 # Linear relationship with some noise # Create a linear regression model model = LinearRegression() model.fit(X, y) # Predict values y_pred = model.predict(X) # Plotting the results plt.scatter(X, y, color='blue', label='Data') plt.plot(X, y_pred, color='red', label='Linear Regression Line') plt.title('Linear Regression') plt.xlabel('X') plt.ylabel('y') plt.legend() plt.show()
